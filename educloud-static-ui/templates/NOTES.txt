Get the application URL by running these commands:

{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "educloud-static-ui.fullname" . }})
  export NODE_IP={{ .Values.hostname }}
  echo curl http://$NODE_IP:$NODE_PORT
{{- else if contains "ClusterIP" .Values.service.type }}
  export NODE_IP={{ .Values.ingress.host }}
  echo curl http://$NODE_IP
{{- end }}
